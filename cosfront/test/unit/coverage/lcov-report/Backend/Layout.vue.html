  methods: {
      logout() {
          var vm=this;
          vm.$axios.get('http://localhost:5000/Account/Logout')
          .then(function(response){
              vm.fullscreenLoading=false;
              if(response.data.status==1){
                  vm.$router.push({
                      name:'Home'
                  });
              }else{
                  vm.$message.error(response.data.message);
              }
          }).catch(function(error){
